Preprocessing
Several columns of the raw dataset contain mostly empty strings or NA values, so we remove them. We
also exclude metadata columns unrelated to motion measurement:
X
,
user_name
,
raw_timestamp_part_1
,
raw_timestamp_part_2, cvtd_timestamp, new_window, and num_window.
# Remove columns with too many missing values but keep 'classe'
valid_cols <- colMeans(is.na(train_raw)) < 0.9
valid_cols["classe"] <- TRUE # ensure target column is kept
train_clean <- train_raw[, valid_cols]
# Remove irrelevant metadata columns
drop_cols <- c("X", "user_name", "raw_timestamp_part_1", "raw_timestamp_part_2",
"cvtd_timestamp", "new_window", "num_window")
train_clean <- train_clean[,
!(names(train_clean) %in% drop_cols)]
# Convert target to factor
train_clean$classe <- as.factor(train_clean$classe)
# Remove near-zero variance predictors
nzv <- nearZeroVar(train_clean)
if (length(nzv) > 0) train_clean <- train_clean[, -nzv]
# Verify target variable

Feature Preprocessing (Imputation, Scaling, PCA)
We apply preprocessing using
caret::preProcess()
to handle missing values, scale the data, and perform
PCA to reduce dimensionality (retaining 90% of variance).
preproc_model <- preProcess(
train_set[,
-ncol(train_set)],
method = c("center", "scale", "knnImpute", "pca"),
thresh = 0.9
)
train_processed <-
predict(preproc_model, train_set[, -ncol(train_set)])
valid_processed <-
predict(preproc_model, valid_set[, -ncol(valid_set)])
Model Training and Evaluation
We train a k-Nearest Neighbors (kNN) model and evaluate its performance using the validation set.
knn_model <- train(
x = train_processed,
y = train_set$classe,
method = "knn"
)
pred_valid <-
predict(knn_model, valid_processed)
conf_matrix <-
confusionMatrix(valid_set$classe, pred_valid)
